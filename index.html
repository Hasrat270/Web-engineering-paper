Essential Syntax Patterns
from abc import ABC, abstractmethod   # For abstract classes
class MyClass(ABC):                    # Inherit from ABC
    @abstractmethod                      # Required decorator
    def method(self): pass

class Child(Parent):                   # Inheritance
    def __init__(self, a, b):
        super().__init__(a, b)           # Call parent constructor

self.__variable = 0                    # Private (double underscore)
global x                               # Access global var inside function
Avoid Common Pitfalls
✗  _init_   →  ✓  __init__   (Double underscores on both sides)
✗  _grade   →  ✓  __grade    (Use 2 underscores for true private)
✗  super(Parent).__init__()
           →  ✓  super().__init__()
✗  Abstract method body
           →  ✓  Always use 'pass' or '...'

Advanced OOP: MRO & Visibility
# MRO (Method Resolution Order)
class C(A, B): pass   # Searches C -> A -> B

# Access Modifiers
self.public = 1      # Accessible everywhere
self._protected = 2   # Internal use (convention)
self.__private = 3    # Name mangling (hard to access)
